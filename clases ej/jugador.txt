public class Futbolista extends IntegrantesSeleccion
{
    private Integer dorsal;
    private String demarcacion;
    
    public Futbolista() {
    }
    
    public Futbolista(final Integer id, final String nombres, final String apellidos, final Integer edad, final Integer dorsal, final String demarcacion) {
        super(id, nombres, apellidos, edad);
        this.dorsal = dorsal;
        this.demarcacion = demarcacion;
    }
    
    public void jugarPartido() {
    }
    
    public void entrenar() {
    }
    
    public Integer getDorsal() {
        return this.dorsal;
    }
    
    public void setDorsal(final Integer dorsal) {
        this.dorsal = dorsal;
    }
    
    public String getDemarcacion() {
        return this.demarcacion;
    }
    
    public void setDemarcacion(final String demarcacion) {
        this.demarcacion = demarcacion;
    }
}
